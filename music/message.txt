 function each(coll, f) { 
    if (Array.isArray(coll)) { 
          for (var i = 0; i < coll.length; i++) { 
                f(coll[i], i); 
          } 
    } else { 
       for (var key in coll) { 
                f(coll[key], key); 
               } 
            } 
         }
         function filter(array, predicate) {
            var acc = [];
            each(array, function(element) {
               if (predicate(element)) {
                  acc.push(element);
               }
            });
            return acc;
         }
